    PathResolver

    file path resolver.

    Example:

    let pathResolver = new PathResolver('./src', 'foo/bar.js', 'foo-bar', 'out/foo/bar.js', 'out');
pathResolver.importPath; // 'foo-bar'
pathResolver.filePath; // 'src/foo/bar.js'
pathResolver.resolve('./baz.js'); // 'src/foo/baz.js'
